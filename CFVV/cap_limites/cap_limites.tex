\section{Limites}

\construirSec

\subsection*{Exercícios resolvidos}

\construirExeresol

\begin{exeresol}
  Calcule $\lim_{(x, y)\to (2, 3)} 2x + y$.
\end{exeresol}
\begin{resol}
  Podemos calcular o limite solicitado da seguinte forma
  \begin{equation}
      \lim_{(x, y)\to (2, 3)} 2x + y = 2\times 2 + 3 = 7.
  \end{equation}
\end{resol}
