\section{Limites no ponto}

\construirSec

\subsection*{Exercícios resolvidos}

\construirExeresol

\begin{exeresol}
  Calcule $\lim_{x\to 2} 2x$.
\end{exeresol}
\begin{resol}
  Podemos calcular o limite solicitado da seguinte forma
  \begin{equation}
    \begin{split}
      \lim_{x\to 2} 2x &= 2\lim_{x\to 2} x\\
      &= 2\times 2 = 4.
    \end{split}
  \end{equation}
\end{resol}
